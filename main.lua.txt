--!strict
local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

-- Create ScreenGui
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "MadoHubUI"
screenGui.ResetOnSpawn = false
screenGui.Parent = playerGui

-- Create background frame
local bg = Instance.new("Frame")
bg.Size = UDim2.new(1,0,1,0)
bg.BackgroundColor3 = Color3.fromRGB(15,15,15)
bg.BackgroundTransparency = 1 -- start invisible
bg.Parent = screenGui

-- Create title label
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1,0,0,100)
title.Position = UDim2.new(0,0,0.3,0)
title.BackgroundTransparency = 1
title.Text = "MadoHub"
title.TextColor3 = Color3.fromRGB(255,255,255)
title.Font = Enum.Font.GothamBlack
title.TextScaled = true
title.TextTransparency = 1 -- start invisible
title.Parent = bg

-- Create subtitle
local subtitle = Instance.new("TextLabel")
subtitle.Size = UDim2.new(1,0,0,50)
subtitle.Position = UDim2.new(0,0,0.45,0)
subtitle.BackgroundTransparency = 1
subtitle.Text = "Choose your language - اختر لغتك"
subtitle.TextColor3 = Color3.fromRGB(255,255,255)
subtitle.Font = Enum.Font.GothamBold
subtitle.TextScaled = true
subtitle.TextTransparency = 1 -- start invisible
subtitle.Parent = bg

-- Create buttons
local function createButton(text, position)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0.3,0,0,50)
    btn.Position = position
    btn.BackgroundColor3 = Color3.fromRGB(35,35,35)
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.Font = Enum.Font.GothamBold
    btn.TextScaled = true
    btn.Text = text
    btn.TextTransparency = 1 -- start invisible
    btn.Parent = bg
    return btn
end

local englishBtn = createButton("English", UDim2.new(0.35,0,0.6,0))
local arabicBtn = createButton("Arabic", UDim2.new(0.35,0,0.7,0))

-- Tween function
local function fadeIn(inst, time)
    TweenService:Create(inst, TweenInfo.new(time), {TextTransparency = 0, BackgroundTransparency = 0}):Play()
end

local function fadeOut(inst, time)
    TweenService:Create(inst, TweenInfo.new(time), {TextTransparency = 1, BackgroundTransparency = 1}):Play()
end

-- Animate UI sequence
TweenService:Create(bg, TweenInfo.new(1.5), {BackgroundTransparency = 0}):Play()
wait(1)
fadeIn(title, 1)
wait(1)
fadeIn(subtitle, 1)
wait(0.5)
fadeIn(englishBtn, 1)
fadeIn(arabicBtn, 1)

-- Button functionality
englishBtn.MouseButton1Click:Connect(function()
    print("English selected")
    -- You can add your English UI loading here
    fadeOut(bg, 1)
    wait(1)
    screenGui:Destroy()
end)

arabicBtn.MouseButton1Click:Connect(function()
    print("Arabic selected")
    -- You can add your Arabic UI loading here
    fadeOut(bg, 1)
    wait(1)
    screenGui:Destroy()
end)
