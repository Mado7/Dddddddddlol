local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

-- ScreenGui
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "MadoHubUI"
screenGui.ResetOnSpawn = false
screenGui.Parent = playerGui

-- Fullscreen Gradient Background
local bg = Instance.new("Frame")
bg.Size = UDim2.new(1, 0, 1, 0)
bg.Position = UDim2.new(0, 0, 0, 0)
bg.BackgroundColor3 = Color3.fromRGB(10, 10, 30)
bg.BackgroundTransparency = 1
bg.Parent = screenGui

local gradient = Instance.new("UIGradient")
gradient.Color = ColorSequence.new{
    ColorSequenceKeypoint.new(0, Color3.fromRGB(10, 10, 30)),
    ColorSequenceKeypoint.new(1, Color3.fromRGB(50, 20, 80))
}
gradient.Rotation = 45
gradient.Parent = bg

-- Tween helper
local function tween(inst, props, time, style, dir)
    local tween = TweenService:Create(inst, TweenInfo.new(time, style or Enum.EasingStyle.Quad, dir or Enum.EasingDirection.Out), props)
    tween:Play()
    return tween
end

-- Buttons
local function createButton(text, posY)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0.35,0,0,50) -- mobile friendly width
    btn.AnchorPoint = Vector2.new(0.5,0)
    btn.Position = UDim2.new(0.5,0,posY,0)
    btn.BackgroundColor3 = Color3.fromRGB(45,45,60)
    btn.BorderSizePixel = 0
    btn.Text = text
    btn.Font = Enum.Font.GothamBold
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.TextScaled = true
    btn.BackgroundTransparency = 0.2
    btn.TextTransparency = 1
    btn.ClipsDescendants = true
    btn.Parent = bg

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0,12)
    corner.Parent = btn

    btn.MouseEnter:Connect(function()
        tween(btn, {BackgroundColor3 = Color3.fromRGB(70,50,120)}, 0.2)
    end)
    btn.MouseLeave:Connect(function()
        tween(btn, {BackgroundColor3 = Color3.fromRGB(45,45,60)}, 0.2)
    end)

    return btn
end

local englishBtn = createButton("English", 0.55)
local arabicBtn = createButton("Arabic", 0.65)

-- Animate UI sequence
tween(bg, {BackgroundTransparency = 0}, 1.5)
wait(0.5)
tween(title, {TextTransparency = 0}, 1.2)
wait(0.3)
tween(subtitle, {TextTransparency = 0}, 1)
wait(0.3)
tween(englishBtn, {TextTransparency = 0}, 0.8)
tween(arabicBtn, {TextTransparency = 0}, 0.8)

-- Buttons click
local function exitUI()
    tween(bg, {BackgroundTransparency = 1}, 0.8)
    wait(0.8)
    screenGui:Destroy()
end

englishBtn.MouseButton1Click:Connect(exitUI)
arabicBtn.MouseButton1Click:Connect(exitUI)

-- DOT BUTTERFLIES
local butterflies = {}
local numDots = 25

for i = 1, numDots do
    local dot = Instance.new("Frame")
    dot.Size = UDim2.new(0,8,0,8)
    dot.Position = UDim2.new(math.random(),0,math.random(),0)
    dot.BackgroundColor3 = Color3.fromRGB(255,200,255)
    dot.BorderSizePixel = 0
    dot.AnchorPoint = Vector2.new(0.5,0.5)
    dot.BackgroundTransparency = 0
    dot.Parent = bg

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0.5,0)
    corner.Parent = dot

    local dx = math.random(-50,50)/100
    local dy = math.random(20,60)/100

    table.insert(butterflies, {dot = dot, dx = dx, dy = dy})
end

RunService.RenderStepped:Connect(function(delta)
    for _, b in pairs(butterflies) do
        local newPos = b.dot.Position + UDim2.new(b.dx*delta,0,b.dy*delta,0)
        if newPos.X.Scale > 1 then newPos = UDim2.new(0,0,newPos.Y.Scale,0) end
        if newPos.X.Scale < 0 then newPos = UDim2.new(1,0,newPos.Y.Scale,0) end
        if newPos.Y.Scale > 1 then newPos = UDim2.new(newPos.X.Scale,0,0,0) end
        b.dot.Position = newPos
    end
end)
